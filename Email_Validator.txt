import re  # It maintains the regular expressions for the different versions of email addresses

def validate_email(email_input):
    # defining the regular expressions for the email addresses
    condition = r"^[a-zA-Z0-9-_]+@[a-zA-Z0-9]+\.[a-z]{1,3}$"  # Corrected the regex

    if re.match(condition, email_input):
        return True
    else:
        return False
    

# User Input
email_input = input("Enter your email address: ")
print(validate_email(email_input))
